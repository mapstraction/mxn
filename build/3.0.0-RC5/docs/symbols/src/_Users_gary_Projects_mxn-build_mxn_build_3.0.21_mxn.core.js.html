414 	if (this.api === api) {
415 		return;

417 
418 	var center = this.getCenter();
419 	var zoom = this.getZoom();

422 	this.currentElement.style.display = 'none';
423

426 	this.currentElement.style.display = 'block';

428 	this.api = api;
429 	this.onload[api] = [];
430 	
431 	if (!this.maps.hasOwnProperty(this.api)) {
432 		init.apply(this);
433 
434 		for (var i = 0; i < this.markers.length; i++) {
435 			this.addMarker(this.markers[i], true);
436 		}
437 
438 		for (var j = 0; j < this.polylines.length; j++) {
439 			this.addPolyline( this.polylines[j], true);
440 		}
441 
442 		this.setCenterAndZoom(center,zoom);		
443 	}
444 	else {
445 
446 		//sync the view
447 		this.setCenterAndZoom(center,zoom);
448 
449 		//TODO synchronize the markers and polylines too
450 		// (any overlays created after api instantiation are not sync'd)
451 	}
452 
453 	this.addControls(this.addControlsArgs);

479 /**

488 /**
489  * Run any queued provider API calls for the methods defined in the provider's implementation.
490  * For example, if defferable in mxn.[provider].core.js is set to {getCenter: true, setCenter: true}
491  * then any calls to map.setCenter or map.getCenter will be queued up in this.onload. When the provider's
492  * implementation loads the map, it calls this.runDeferred and any queued calls will be run.
493  */
494 Mapstraction.prototype.runDeferred = function(){
495 	while(this.onload[this.api].length > 0) {  
496 		this.onload[this.api].shift().apply(this); //run deferred calls
497 	}
498 };

520 /**
521  * Add a listener for an event.
522  * @param {string} type Event type to attach listener to
523  * @param {Function} func Callback function
524  * @param {Object} caller Callback object
525  */
526 Mapstraction.prototype.addEventListener = function() {
527 	var listener = {};
528 	listener.event_type = arguments[0];
529 	listener.callback_function = arguments[1];
530 
531 	// added the calling object so we can retain scope of callback function
532 	if(arguments.length == 3) {
533 		listener.back_compat_mode = false;
534 		listener.callback_object = arguments[2];
535 	}
536 	else {
537 		listener.back_compat_mode = true;
538 		listener.callback_object = null;
539 	}
540 	this.eventListeners.push(listener);
541 };
542 
543 /**
544  * Call listeners for a particular event.
545  * @param {string} sEventType Call listeners of this event type
546  * @param {Object} oEventArgs Event args object to pass back to the callback
547  */
548 Mapstraction.prototype.callEventListeners = function(sEventType, oEventArgs) {
549 	oEventArgs.source = this;
550 	for(var i = 0; i < this.eventListeners.length; i++) {
551 		var evLi = this.eventListeners[i];
552 		if(evLi.event_type == sEventType) {
553 			// only two cases for this, click and move
554 			if(evLi.back_compat_mode) {
555 				if(evLi.event_type == 'click') {
556 					evLi.callback_function(oEventArgs.location);
557 				}
558 				else {
559 					evLi.callback_function();
560 				}
561 			}
562 			else {
563 				var scope = evLi.callback_object || this;
564 				evLi.callback_function.call(scope, oEventArgs);
565 			}
566 		}
567 	}
568 };

655 		}

1301 /**
1302  * Returns the distance in kilometers between two <code>mxn.LatLonPoint</code> objects.
1303  * @param {mxn.LatLonPoint} otherPoint The other point to measure the distance from to this one
1304  * @returns The distance between the points in kilometers
1305  * @type {number}
1306  */
1307 LatLonPoint.prototype.distance = function(otherPoint) {
1308 	// Uses Haversine formula from http://www.movable-type.co.uk
1309 	var rads = Math.PI / 180;
1310 	var diffLat = (this.lat-otherPoint.lat) * rads;
1311 	var diffLon = (this.lon-otherPoint.lon) * rads; 
1312 	var a = Math.sin(diffLat / 2) * Math.sin(diffLat / 2) +
1313 		Math.cos(this.lat*rads) * Math.cos(otherPoint.lat*rads) * 
1314 		Math.sin(diffLon/2) * Math.sin(diffLon/2); 
1315 	return 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a)) * 6371; // Earth's mean radius in km
1316 };

1993  * @param {number} tolerance The simplification tolerance; 1.0 is a good starting point
1994  */
1995 Polyline.prototype.simplify = function(tolerance) {
1996 	var reduced = [];
1997 
1998 	// First point
1999 	reduced[0] = this.points[0];
2000 
2001 	var markerPoint = 0;
2002 
2003 	for (var i = 1; i < this.points.length-1; i++){
2004 		if (this.points[i].distance(this.points[markerPoint]) >= tolerance)
2005 		{
2006 			reduced[reduced.length] = this.points[i];
2007 			markerPoint = i;
2008 		}
2009 	}
2010 
2011 	// Last point
2012 	reduced[reduced.length] = this.points[this.points.length-1];
2013 
2014 	// Revert
2015 	this.points = reduced;

2022 /**
2023  * Creates a Mapstraction Radius for drawing circles around a given point. Note that creating
2024  * a radius performs a lot of initial calculation which can lead to increased page load times.
2025  * @name mxn.Radius
2026  * @constructor
2027  * @param {mxn.LatLonPoint} center Central <code>mxn.LatLonPoint</code> of the radius
2028  * @param {number} quality Number of points that comprise the approximated circle (20 is a good starting point)
2029  * @exports Radius as mxn.Radius
2030  */
2031 var Radius = mxn.Radius = function(center, quality) {
2032 	this.center = center;
2033 	var latConv = center.latConv();
2034 	var lonConv = center.lonConv();
2035 
2036 	// Create Radian conversion constant
2037 	var rad = Math.PI / 180;
2038 	this.calcs = [];
2039 
2040 	for(var i = 0; i < 360; i += quality){
2041 		this.calcs.push([Math.cos(i * rad) / latConv, Math.sin(i * rad) / lonConv]);
2042 	}
2043 };

2045 /**
2046  * Returns the <code>mxn.Polyline</code> of a circle around the point based on a new radius value.
2047  * @param {number} radius The new radius value
2048  * @param {string} color RGB fill color expressed in the form <code>#RRGGBB</code>
2049  * @returns {Polyline} The calculated <code>mxn.Polyline</code>
2050  */
2051 Radius.prototype.getPolyline = function(radius, color) {
2052 	var points = [];
2053 
2054 	for(var i = 0; i < this.calcs.length; i++){
2055 		var point = new LatLonPoint(
2056 			this.center.lat + (radius * this.calcs[i][0]),
2057 			this.center.lon + (radius * this.calcs[i][1])
2058 		);
2059 		points.push(point);
2060 	}
2061 	
2062 	// Add first point
2063 	points.push(points[0]);
2064 
2065 	var line = new Polyline(points);
2066 	line.setColor(color);
2067 
2068 	return line;
2069 };
